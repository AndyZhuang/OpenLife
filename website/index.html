Quick Install

# 1. Install ZeroClaw (required)
cargo install zeroclaw

# 2. Clone and build OpenLife
git clone https://github.com/openlife-ai/openlife.git
cd openlife
cargo build --release

# 3. Add to PATH
sudo cp target/release/openlife /usr/local/bin/

# 4. Configure API key
openlife config set --provider openai --api-key YOUR_KEY

# 5. Run!
openlife bio list
openlife agent
openlife gateway
